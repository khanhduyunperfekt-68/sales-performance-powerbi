Total Profit = SUM ( sale_data[Profit] )

Total Orders = DISTINCTCOUNT ( sale_data[Order ID] )

Total Customers = DISTINCTCOUNT ( sale_data[Customer ID] )

AOV = DIVIDE([Total Sales],[Total Orders])

Margin = DIVIDE ( [Total Profit], [Total Sales] )

Profit LY = 
CALCULATE ( [Total Profit], SAMEPERIODLASTYEAR ( 'Date'[Date] ) )

Profit margin = DIVIDE([Total Profit],[Total Sales])

Profit PM = 
CALCULATE ( [Total Profit], DATEADD ( 'Date'[Date], -1, MONTH ) )

Profit Rank = 
RANKX (
    ALL ( sale_data[Product Name] ),
    [Total Profit],
    ,
    DESC
)

Profit MoM % = 
DIVIDE ( [Total Profit] - [Profit PM], [Profit PM] )
